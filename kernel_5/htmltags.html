	: <html>  ( -- )
		BEGIN                                    
			BEGIN
				BL (word) DUP  
			WHILE                                
				2DUP  S" </html>"  COMPARE 
				0= IF                             
					2DROP                        
					EXIT
				THEN                              
				2DUP  S" <code>"  COMPARE 
				0= IF                               
					2DROP                        
					EXIT
				THEN                              
				2DROP
			REPEAT  2DROP                           
		REFILL 0=  UNTIL                          
	;  IMMEDIATE
