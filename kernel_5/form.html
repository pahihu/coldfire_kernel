    : .title  ( --) 'TITLE @execute ;
    
    : new_page ( --)
    	?page PAGE  
		@days .days 
		$2 SPACES  @ticks .time
    	."        KERNEL VERSION : " .version
    	line# character_max @ $10 - TAB ." PAGE: " page#  .
    	CR .HEADING
    	CR .title
    	CR BEGIN 
			[CHAR] _ EMIT character#  character_max @ 2- > 
		UNTIL CR send 
	;
    
  
    
	: new_form ( --)
		one page!  
		TRUE first_page C!  
	;
    
	: first_page 
		new_form 
		new_page 
	;

    : form   ( --)
		CR line# line_max @ < not IF ( end of page)
			new_page 
		THEN 
	;
